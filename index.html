Código JavaScript para verificação de keys:

class KeyAuth {
    constructor(apiUrl) {
        this.apiUrl = apiUrl;
    }
    
    async verifyKey(key, hwid = null) {
        const formData = new FormData();
        formData.append('action', 'verify');
        formData.append('key', key);
        if (hwid) formData.append('hwid', hwid);
        
        try {
            const response = await fetch(this.apiUrl, {
                method: 'POST',
                body: formData
            });
            const result = await response.json();
            return result.success || false;
        } catch {
            return false;
        }
    }
}

// Uso
const auth = new KeyAuth('https://seusite.com/api.php');
auth.verifyKey('GHOST-1234-5678-9012').then(valid => {
    if (valid) {
        console.log('✅ Acesso autorizado!');
    } else {
        console.log('❌ Acesso negado!');
    }
});

# Criar keys (requer senha de admin)
curl -X POST "https://seusite.com/api.php" \
     -d "action=create&admin_password=GHOSTDEVOWNER_2025/2000_LOL&key_type=monthly&quantity=5"

# Listar keys (requer senha de admin)
curl -X POST "https://seusite.com/api.php" \
     -d "action=list&admin_password=GHOSTDEVOWNER_2025/2000_LOL"

# Obter estatísticas
curl -X POST "https://seusite.com/api.php" \
     -d "action=stats"
📋 Endpoints da API:
• verify - Verificar key
• create - Criar keys (requer senha admin)
• list - Listar keys (requer senha admin)
• edit - Editar key (requer senha admin)
• disable - Desativar key (requer senha admin)
• delete - Excluir key (requer senha admin)
• stats - Estatísticas públicas
